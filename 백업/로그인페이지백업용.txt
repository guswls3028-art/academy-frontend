import { useState, useRef, useEffect } from "react";


import Landing from "./pages/auth/Landing";

function App() {
  return <Landing />;
}

export default App;





export default function Landing() {
  const [open, setOpen] = useState(false);
  const ref = useRef<HTMLDivElement>(null);

  // close when clicking outside
  useEffect(() => {
    function handler(e: MouseEvent) {
      if (ref.current && !ref.current.contains(e.target as Node)) {
        setOpen(false);
      }
    }
    if (open) document.addEventListener("mousedown", handler);
    return () => document.removeEventListener("mousedown", handler);
  }, [open]);

  return (
    <div className="min-h-screen flex flex-col items-center justify-center bg-[#050B1B] text-white select-none">

      {/* TITLE SECTION (animated) */}
  <div
    className={`
      flex flex-col items-center transition-all duration-500
      ${open ? "translate-y-12" : "translate-y-40"}
    `}
  >
        <h1 className="text-8xl font-extrabold tracking-wide mb-2">
          철과 함께
        </h1>

        {/* Fe BELOW "철" */}
        <div className="relative w-full flex justify-center">
          <span className="absolute left-[calc(23%-4.5rem)] text-5xl text-gray-400 font-semibold">
            Fe
          </span>
        </div>

        <p className="text-gray-400 text-sm mt-6">
          현진이가 만든 임시 로그인 화면
        </p>
      </div>

      {/* FIXED PANEL AREA */}
      <div className="relative mt-20 h-[280px] w-full flex justify-center items-start">

        {/* BUTTONS */}
        <div
          className={`
            flex gap-6 transition-all duration-500
            ${open ? "mt-12 opacity-0 scale-95 pointer-events-none" : "mt-30 opacity-100 scale-100"}
          `}
        >
          <button
            className="px-9 py-3 text-2xl font-bold bg-[#1b2655] rounded-md shadow-lg hover:opacity-90 transition"
            onClick={() => setOpen(true)}
          >
            로그인
          </button>

          <button className="px-9 py-3 text-2xl font-bold bg-[#35343B] rounded-md shadow-lg hover:opacity-90 transition">
            게시판
          </button>
        </div>

        {/* LOGIN FORM */}
        <div
          ref={ref}
          className={`
            absolute w-[500px] bg-[#1b2655] rounded-xl shadow-xl p-8 transition-all duration-500
            ${open ? "opacity-100 scale-100" : "opacity-0 scale-90 pointer-events-none"}
          `}
        >
          <div className="flex flex-col gap-4">

            <input
              className="px-4 py-3 rounded-lg bg-white text-black placeholder-gray-400 focus:outline-none"
              placeholder="아이디"
            />

            <input
              className="px-4 py-3 rounded-lg bg-white text-black placeholder-gray-400 focus:outline-none"
              placeholder="비밀번호"
            />

            <button className="bg-white text-[#1b2655] font-bold py-3 rounded-lg hover:bg-gray-200 transition">
              로그인
            </button>

            <p className="text-center text-sm text-gray-300 mt-2 hover:underline cursor-pointer">
              비번 까먹었음? ㅋㅋ 누르셈
            </p>

          </div>
        </div>

      </div>
    </div>
  );
}
